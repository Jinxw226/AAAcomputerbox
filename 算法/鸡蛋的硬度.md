#鸡蛋的硬度 #飞机最低可俯冲速度
# 方式一

![[鸡蛋硬度方式1.png]]

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N =110 , M = 11;
int n,m;
int f[N][M];

int main(){
  while(cin>>n>>m){
    for (int i =1;i <= n;i++) f[i][1] = i;
    for (int i =1;i <= m;i++) f[1][i] = 1;

    for(int i =2;i <= n; i++){
      for(int j = 2;j <= m;j++){
        f[i][j] = f[i][j-1];
        for(int k = 1; k <= i; k++){
          f[i][j] = min(f[i][j],max(f[k-1][j-1],f[i-k][j]) + 1);
        }
      }
    }
    cout<<f[n][m]<<endl;
  }
  return 0;
}
```

时间复杂度O(n) = n方m

# 方式二

![[鸡蛋硬度方式2.png]]

时间复杂度O(n)=nm

```cpp
#include<cstdio>

const int N = 101,M = 11;
int n,m;
int f[N][N];

int main()
{
  while(~scanf("%d%d",&n,&m)){
    for(int i = 1; i <= n; i++){
      for(int j = 1;j <=m; j++){
        f[i][j] = f[i-1][j] + f[i-1][j-1] + 1;
      }
      if(f[i][m] >= n){
        printf("%d\n",i);
        break;
      }
    }
  }
  return 0;
}
```
