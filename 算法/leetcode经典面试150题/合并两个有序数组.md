两个非递减数组排序
第一个数组设置为nums1，内含有m个数据，并且长度设置为m+n,便于直接将合并的数据存于nums1
第二个数组设置为nums2，内含有n个数据

```cpp
#include<vector>
#include<iostream>
using namespace std;

class Solution {
  public:
  void merge(vector<int>& nums1, int m , vector<int>& nums2, int n){
    int p1 = m -1;
    int p2 = n -1;
    int sum = m + n -1;
    while (p1 >= 0  && p2 >= 0)
    {
      if(nums1[p1] > nums2[p2]){
        nums1[sum] = nums1[p1];
        p1--;
      }
      else{
        nums1[sum] = nums2[p2];
        p2--;
      }
      sum--;
    }
    while(p2 >= 0 ){
      nums1[sum] = nums2[p2];
      p2--;
      sum--;
    }
    for(int i = 0 ; i < nums1.size(); i++){
      cout << nums1[i] << " " ;
    }
  }
};
int main(){
  // 方式一
  // vector<int> nums1 = {1,2,3,0,0,0};
  // vector<int> nums2 = {2,5,6};
  // Solution().merge(nums1,3,nums2,3);
  // 方式二
  // vector<int> nums1 = {0};
  // vector<int> nums2 = {1};
  // Solution().merge(nums1,0,nums2,1);
  // 方式三
  vector<int> nums1 = {1};
  vector<int> nums2 = {0};
  Solution().merge(nums1,1,nums2,0);
}
```
